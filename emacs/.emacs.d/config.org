** org mode

#+BEGIN_SRC emacs-lisp

(require 'org)

;; set a time-stamp when todo state changes
(setq org-log-done 'note)

;; added custom states for todo list
(setq org-todo-keywords '((sequence "TODO(t)" "|" "DONE(d)")
			  (sequence "|" "CANCEL(c)")
                          (sequence "STARTED(s)" "|")
                          (sequence "WAITING(w)" "|" )))

;; added color for custom states
(setq org-todo-keyword-faces
      (quote (("CANCEL" :foreground "red" :weight bold)
	      ("DONE" :foreground "green" :weight bold)
	      ("WAITING" :foreground "yellow")
              ("STARTED" :foreground "orange" :weight bold))))


;; do a separate add-to-list to add new file extentions
;; put default mode for specific files
(add-to-list 'auto-mode-alist
	     '("\\.note" . org-mode))


;; default directory to save org mode capture files
(setq org-directory "/home/sohom/work_space/todo/")
(setq org-default-notes-file (concat org-directory "/TODO.org"))
(define-key global-map "\C-cc" 'org-capture)

(setq org-capture-templates
 '(("t" "Todo" entry (file+headline "~/work_space/todo/TODO.org" "TASKS")
        "* TODO %?\n  %i\n  %a")
   ("b" "Blog" entry (file+headline "~/work_space/todo/TODO.org" "BLOG")
	"* TODO %?\n  %i\n  %a")
   ("p" "Project" entry (file+headline "~/work_space/todo/TODO.org" "PROJECTS")
	"* TODO %?\n  %i\n  %a")))


;; TODO file
(global-set-key (kbd "<f6>") (lambda() (interactive)(find-file "/home/sohom/work_space/todo/TODO.org")))


;; agenda file 
(add-to-list 'org-agenda-files "/home/sohom/work_space/todo/TODO.org")
(global-set-key "\C-ca" 'org-agenda)
(setq org-agenda-use-time-grid t)

;; tags

(setq org-tag-alist '(("work" . ?w)
		      ("study" . ?s)
                      ("fsmk" . ?f)
		      ("family" . ?y)))

;; org refile
(custom-set-variables '(org-refile-targets (quote (("todoArchive.org" :maxlevel . 1)))))


#+END_SRC
